def self.from_json(api, json)
  Form.new(
      api,
      json['name'],
      Hash[json['fields'].map { |k2, field|
        [k2, Field.new(field['type'], field['default'], k2 == 'q')]
      }],
      json['method'],
      json['rel'],
      json['enctype'],
      json['action']
  )
end
