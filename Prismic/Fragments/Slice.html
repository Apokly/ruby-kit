def as_html(link_resolver=nil)
  classes = ['slice']
  unless (@slice_label.nil?)
    classes.push(@slice_label)
  end
  %[<div data-slicetype="#{@slice_type}" class="#{classes.join(' ')}">#{@value.as_html(link_resolver)}</div>]
end
